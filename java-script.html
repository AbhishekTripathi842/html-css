PROMISES :-

back
    -promises are used to handle asynchronous operation in java script.
    -promises are special java script object
    -promises have two states
    -success (resolve)
    -failure (reject)
    - promises can be created using promise class.
    -promise is the predefine class in java script.
    -promises can be consumed using then() function.

    • Examples

    examp 1 =>

    let my_promise = new Promise((resolve,reject)=>{
    resolve("tommorow i will be at home")
    })
    my_promise.then((posres)=>{
    console.log(posres)
    },(err)=>{console.log(err)})

    examp 2 =>

    let my_promise = new Promise((resolve,reject)=>{
    setTimeout(()=>{
    resolve("tommorow i will be at home")
    },5000)
    })
    my_promise.then((posres)=>{
    console.log(posres)
    },(err)=>{console.log(err)})

    examp 3 =>

    let my_mypromise = new Promise((resolve,reject)=>{
    reject("fail")
    resolve("success")
    })
    my_mypromise.then((posres)=>{
    console.log("pos",posres)
    },(err)=>{
    console.log("neg",err)
    })

    examp 4 =>

    let my_mypromise = new Promise((resolve,reject)=>{
    setTimeout(()=>{
    resolve("success")
    },5000)
    setTimeout(()=>{
    reject("fail")
    },4000)
    })
    my_mypromise.then((posres)=>{
    console.log(posres)
    },(error)=>{
    console.log(error)
    })

Solve callback-hell problem with async and await :-

back

    • Examples

    function add(num)
    {
    return new Promise((resolve, reject)=>{
    resolve(num + 5)
    })
    }
    function sub(num)
    {
    return new Promise((resolve,reject)=>{
    resolve(num - 3)
    })
    }
    function mul(num)
    {
    return new Promise((resolve, reject)=>{
    resolve(num * 4)
    })
    }
    function div(num)
    {
    return new Promise((resolve, reject)=>{
    resolve(num / 2)
    })
    }
    async function myFun()
    {
    let addRes = await add(5)
    let subRes = await sub(addRes)
    let mulRes = await mul(subRes)
    let divRes = await div(mulRes)
    console.log(divRes)
    }

    myFun()
